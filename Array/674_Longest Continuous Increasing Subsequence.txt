class Solution {
    public int findLengthOfLCIS(int[] nums) {
        if(nums.length==0)
            return 0;
        
        int max=1;
        for(int i=0;i<nums.length-1;i++){
            int count=1;
            while((i<nums.length-1)&&(nums[i+1]>nums[i])){
                count++;
                i++;
            }
            if(count>max)
                max=count;
            
        }
        return max;
    }
}
